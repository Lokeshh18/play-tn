# Play TN — 100% Free-Tier Full-Stack Build Plan (Supabase Edition)

This document outlines the step-by-step roadmap, database schema, and deployment architecture to convert the static **Play TN** frontend into a production-ready, full-stack application using **Supabase** for database and file storage.

---

## 1. Free-Tier Infrastructure Architecture

Since budget is $0, we must avoid any service that requires a credit card or has time-limited trials. This stack is **permanently free** within reasonable usage limits.

| Service | Provider | Free Tier Details | Role in Play TN |
| :--- | :--- | :--- | :--- |
| **Frontend Hosting** | **Vercel** or **Netlify** | Unlimited bandwidth, custom domains, free HTTPS. | Serves the HTML, CSS, and JS files. |
| **Backend Hosting** | **Render** or **Koyeb** | Free web service instance (750 hours/month). Spins down after 15 mins of inactivity. | Runs the Spring Boot REST API application. |
| **Database** | **Supabase (PostgreSQL)** | 500MB database space, up to 50,000 monthly active users. | Stores user, tournament, registration, and leaderboard data. |
| **Image & File Hosting**| **Supabase Storage** | 1GB free storage space, 2GB monthly bandwidth. | Stores tournament posters, banners, rules PDFs, and logos. |
| **Email & OTP Delivery**| **Resend** | 3,000 emails/month (using `onboarding@resend.dev`). | Sends OTPs and notifications (restricted to account email for testing; bypass for demo). |

## 2. Step 1: Database Schema Design (PostgreSQL)

Create the tables in your free Supabase PostgreSQL database using the following normalized schema. Run this script in the **SQL Editor** tab of your Supabase Dashboard:

```sql
-- 1. Users & Auth
CREATE TABLE users (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    password VARCHAR(255) NOT NULL,
    phone VARCHAR(20),
    role VARCHAR(20) NOT NULL, -- 'PLAYER', 'COACH', 'ORGANISER', 'ADMIN'
    is_verified BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE verification_tokens (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    token VARCHAR(6) NOT NULL, -- 6 digit OTP
    email VARCHAR(100) NOT NULL,
    expiry_date TIMESTAMP NOT NULL
);

-- 2. Profiles
CREATE TABLE player_profiles (
    id BIGINT PRIMARY KEY REFERENCES users(id) ON DELETE CASCADE,
    age INT,
    dob DATE,
    gender VARCHAR(10),
    district VARCHAR(50),
    address TEXT,
    preferred_sport VARCHAR(50),
    experience VARCHAR(50),
    profile_pic VARCHAR(255),
    bio TEXT,
    achievements TEXT
);

CREATE TABLE coach_profiles (
    id BIGINT PRIMARY KEY REFERENCES users(id) ON DELETE CASCADE,
    certifications TEXT,
    specialization VARCHAR(100),
    profile_pic VARCHAR(255)
);

CREATE TABLE organiser_profiles (
    id BIGINT PRIMARY KEY REFERENCES users(id) ON DELETE CASCADE,
    organisation_name VARCHAR(100),
    contact_phone VARCHAR(20),
    logo VARCHAR(255),
    is_approved BOOLEAN DEFAULT FALSE
);

-- 3. Tournaments & Events
CREATE TABLE tournaments (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    name VARCHAR(150) NOT NULL,
    description TEXT,
    sport VARCHAR(50) NOT NULL,
    organiser_id BIGINT NOT NULL REFERENCES organiser_profiles(id),
    poster_url VARCHAR(255),
    rules_pdf_url VARCHAR(255),
    format VARCHAR(50), -- 'SINGLE_ELIMINATION', 'LEAGUE', 'ROUND_ROBIN'
    district VARCHAR(50) NOT NULL,
    venue TEXT NOT NULL,
    maps_location VARCHAR(255),
    registration_start TIMESTAMP WITH TIME ZONE NOT NULL,
    registration_end TIMESTAMP WITH TIME ZONE NOT NULL,
    tournament_start TIMESTAMP WITH TIME ZONE NOT NULL,
    tournament_end TIMESTAMP WITH TIME ZONE NOT NULL,
    entry_fee DECIMAL(10, 2) DEFAULT 0.00,
    prize_pool DECIMAL(10, 2) DEFAULT 0.00,
    min_teams INT,
    max_teams INT,
    status VARCHAR(20) DEFAULT 'DRAFT', -- 'DRAFT', 'SUBMITTED', 'APPROVED', 'LIVE', 'COMPLETED'
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- 4. Teams & Rosters
CREATE TABLE teams (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    captain_id BIGINT NOT NULL REFERENCES player_profiles(id),
    vice_captain_id BIGINT REFERENCES player_profiles(id),
    team_code VARCHAR(10) UNIQUE NOT NULL,
    logo_url VARCHAR(255),
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE team_members (
    team_id BIGINT NOT NULL REFERENCES teams(id) ON DELETE CASCADE,
    player_id BIGINT NOT NULL REFERENCES player_profiles(id) ON DELETE CASCADE,
    status VARCHAR(20) DEFAULT 'PENDING', -- 'PENDING', 'ACCEPTED'
    PRIMARY KEY (team_id, player_id)
);

-- 5. Registrations & Fixtures
CREATE TABLE tournament_registrations (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    tournament_id BIGINT NOT NULL REFERENCES tournaments(id) ON DELETE CASCADE,
    team_id BIGINT REFERENCES teams(id) ON DELETE CASCADE,
    player_id BIGINT REFERENCES player_profiles(id) ON DELETE CASCADE,
    registered_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    payment_status VARCHAR(20) DEFAULT 'PENDING' -- 'PENDING', 'PAID', 'FREE'
);

CREATE TABLE matches (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    tournament_id BIGINT NOT NULL REFERENCES tournaments(id) ON DELETE CASCADE,
    team1_id BIGINT REFERENCES teams(id),
    team2_id BIGINT REFERENCES teams(id),
    player1_id BIGINT REFERENCES player_profiles(id),
    player2_id BIGINT REFERENCES player_profiles(id),
    match_date TIMESTAMP WITH TIME ZONE NOT NULL,
    round_number INT DEFAULT 1,
    status VARCHAR(20) DEFAULT 'SCHEDULED', -- 'SCHEDULED', 'LIVE', 'COMPLETED'
    winner_team_id BIGINT REFERENCES teams(id),
    winner_player_id BIGINT REFERENCES player_profiles(id),
    score_summary VARCHAR(100)
);

-- 6. Leaderboards & Analytics
CREATE TABLE leaderboards (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    player_id BIGINT NOT NULL REFERENCES player_profiles(id) ON DELETE CASCADE,
    sport VARCHAR(50) NOT NULL,
    district VARCHAR(50) NOT NULL,
    matches_played INT DEFAULT 0,
    matches_won INT DEFAULT 0,
    points INT DEFAULT 0,
    trend VARCHAR(5) DEFAULT 'eq' -- 'up', 'dn', 'eq'
);
```

## 4. Step 3: Integrating Supabase Storage (Backend Code)

Instead of using an external library like Cloudinary, we can communicate directly with **Supabase Storage** using standard Spring REST communication tools.

### 1. Supabase Dashboard Setup
1. Go to your **Supabase Dashboard** -> **Storage**.
2. Create a new Bucket named `playtn-media`.
3. Set the bucket to **Public** so uploaded files (like logos and posters) are accessible via a direct public URL.

### 2. Spring Service Implementation
Add this service class to handle file uploads to Supabase Storage:

```java
import org.springframework.beans.factory.annotation.Value;
import org.springframework.http.*;
import org.springframework.stereotype.Service;
import org.springframework.web.client.RestTemplate;
import org.springframework.web.multipart.MultipartFile;
import java.io.IOException;
import java.util.UUID;

@Service
public class SupabaseStorageService {

    @Value("${supabase.url}")
    private String supabaseUrl;

    @Value("${supabase.key}")
    private String supabaseKey;

    @Value("${supabase.bucket:playtn-media}")
    private String bucketName;

    private final RestTemplate restTemplate = new RestTemplate();

    public String uploadFile(MultipartFile file, String folder) throws IOException {
        // Generate a unique filename
        String originalName = file.getOriginalFilename();
        String extension = originalName != null && originalName.contains(".") 
            ? originalName.substring(originalName.lastIndexOf(".")) 
            : "";
        String fileName = folder + "/" + UUID.randomUUID().toString() + extension;

        // Construct API URL
        String uploadUrl = String.format("%s/storage/v1/object/%s/%s", supabaseUrl, bucketName, fileName);

        // Prepare Headers
        HttpHeaders headers = new HttpHeaders();
        headers.set("Authorization", "Bearer " + supabaseKey);
        headers.set("apikey", supabaseKey);
        headers.setContentType(MediaType.valueOf(file.getContentType()));

        HttpEntity<byte[]> requestEntity = new HttpEntity<>(file.getBytes(), headers);

        // Send request to Supabase REST Storage API
        ResponseEntity<String> response = restTemplate.exchange(
            uploadUrl, 
            HttpMethod.POST, 
            requestEntity, 
            String.class
        );

        if (response.getStatusCode() == HttpStatus.OK) {
            // Return public URL of the uploaded object
            return String.format("%s/storage/v1/object/public/%s/%s", supabaseUrl, bucketName, fileName);
        } else {
            throw new RuntimeException("Failed to upload file to Supabase: " + response.getBody());
        }
    }
}
```

Add these corresponding properties to your configuration:
```properties
# Supabase Storage Details
supabase.url=https://your-project-id.supabase.co
supabase.key=your-supabase-service-role-key-or-anon-key
```

### 3. Resend Email Service Implementation
Add this service class to handle OTP email delivery via Resend's REST API. It includes a fallback logic so recruiters testing the resume project can verify their login instantly without domain errors.

```java
import org.springframework.beans.factory.annotation.Value;
import org.springframework.http.*;
import org.springframework.stereotype.Service;
import org.springframework.web.client.RestTemplate;
import java.util.HashMap;
import java.util.Map;

@Service
public class ResendEmailService {

    @Value("${resend.api.key}")
    private String resendApiKey;

    @Value("${resend.from:onboarding@resend.dev}")
    private String fromEmail;

    @Value("${app.demo-mode:true}")
    private boolean demoMode;

    private final RestTemplate restTemplate = new RestTemplate();

    public void sendOtp(String toEmail, String otp) {
        System.out.println("--------------------------------------------------");
        System.out.println("OTP CODE GENERATED FOR " + toEmail + ": " + otp);
        System.out.println("--------------------------------------------------");

        // Resend free tier on onboarding@resend.dev only allows sending to the registered account owner.
        // In Demo/Resume mode, print the OTP to logs and skip the API call for unverified emails to avoid API errors.
        boolean isOwnerEmail = toEmail.equalsIgnoreCase("YOUR_RESEND_REGISTERED_EMAIL@example.com"); // Replace with your Resend email

        if (demoMode && !isOwnerEmail) {
            System.out.println("Demo Mode Active: printed OTP to console. Skipping Resend API call to prevent recipient blocking.");
            return;
        }

        try {
            String url = "https://api.resend.com/emails";

            HttpHeaders headers = new HttpHeaders();
            headers.set("Authorization", "Bearer " + resendApiKey);
            headers.setContentType(MediaType.APPLICATION_JSON);

            Map<String, Object> body = new HashMap<>();
            body.put("from", fromEmail);
            body.put("to", toEmail);
            body.put("subject", "Play TN - Verification OTP");
            body.put("html", String.format(
                "<h3>Welcome to Play TN!</h3>" +
                "<p>Your email verification OTP code is: <strong>%s</strong></p>" +
                "<p><em>Note: If you are a recruiter reviewing this live project, you can bypass this by entering the fallback OTP: <strong>123456</strong>.</em></p>", 
                otp
            ));

            HttpEntity<Map<String, Object>> requestEntity = new HttpEntity<>(body, headers);
            ResponseEntity<String> response = restTemplate.exchange(url, HttpMethod.POST, requestEntity, String.class);

            if (response.getStatusCode() != HttpStatus.OK && response.getStatusCode() != HttpStatus.CREATED) {
                System.err.println("Failed to send email via Resend API: " + response.getBody());
            }
        } catch (Exception e) {
            System.err.println("Resend API error occurred: " + e.getMessage());
        }
    }
}
```

### API Integration Core (`js/api.js`)
Use this standard script to handle API requests and security tokens:
```javascript
const API_BASE_URL = window.location.hostname === 'localhost' || window.location.hostname === '127.0.0.1'
    ? 'http://localhost:8080/api/v1'
    : 'https://play-tn-backend.onrender.com/api/v1'; // Replace with Render URL once deployed

// Fetch wrapper that automatically includes the JWT Token
async function fetchWithAuth(endpoint, options = {}) {
    const token = localStorage.getItem('token');
    
    const headers = {
        'Content-Type': 'application/json',
        ...options.headers
    };
    
    if (token) {
        headers['Authorization'] = `Bearer ${token}`;
    }
    
    const response = await fetch(`${API_BASE_URL}${endpoint}`, {
        ...options,
        headers
    });
    
    if (response.status === 401 || response.status === 403) {
        // Session expired or unauthorized -> redirect to login
        localStorage.removeItem('token');
        window.location.href = '/login.html';
    }
    
    return response;
}
```
